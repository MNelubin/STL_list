399 iterator erase(const_iterator pos) {
400 if (pos == cend() || pos.get_node() == nullptr || pos.get_node() == head || pos.get_node() == tail) {
401 return end();
402 }
403
404 Node* node_to_delete = const_cast<Node*>(pos.get_node());
405
406 std::vector<Node*> update_vec(MAX_LEVEL, nullptr);
407 Node* current_search = head;
408
409 // Find predecessors for the specific node to delete
410 for (int i = current_max_level; i >= 0; --i) {
411 while (current_search && static_cast<size_t>(i) < current_search->forward_pointers.size() &&
412 current_search->forward_pointers[i] != tail &&
413 current_search->forward_pointers[i] != node_to_delete &&
414 (current_search->forward_pointers[i]->value < node_to_delete->value ||
415 (current_search->forward_pointers[i]->value == node_to_delete->value &&
416 current_search->forward_pointers[i] != node_to_delete)
417 )) {
418 current_search = current_search->forward_pointers[i];
419 }
420 if (static_cast<size_t>(i) < update_vec.size()) {
421 update_vec[i] = current_search;
422 } else {
423 assert(false && "update_vec out of bounds in erase predecessor search");
424 }
425 }
426
427 assert(node_to_delete != nullptr && "erase(pos): node_to_delete is null!");
428 assert(update_vec[0] != nullptr && "erase(pos): update_vec[0] (predecessor at L0) is null!");
429 if (update_vec[0]) {
430 assert(static_cast<size_t>(0) < update_vec[0]->forward_pointers.size() && "erase(pos): update_vec[0] has no forward_pointers at L0!");
431 }
432
433 // Critical check: if this fails, node was not properly found or list is inconsistent
434 if (!update_vec[0] ||
435 static_cast<size_t>(0) >= update_vec[0]->forward_pointers.size() ||
436 update_vec[0]->forward_pointers[0] != node_to_delete) {
437 return end(); // Node not found via predecessor scan, possibly inconsistent state or bad iterator
438 }
439
440 Node* successor_node_ptr = node_to_delete->forward_pointers[0];
441 iterator next_iter(successor_node_ptr);
442
443 // Unlink forward pointers - THIS IS THE CORRECTED LOOP
444 int node_actual_level = static_cast<int>(node_to_delete->forward_pointers.size()) - 1;
445 for (int i = 0; i <= node_actual_level; ++i) {
446 if (static_cast<size_t>(i) < update_vec.size() && update_vec[i] &&
447 static_cast<size_t>(i) < update_vec[i]->forward_pointers.size() &&
448 update_vec[i]->forward_pointers[i] == node_to_delete) {
449 // Accessing node_to_delete->forward_pointers[i] is safe now because i <= node_actual_level
450 update_vec[i]->forward_pointers[i] = node_to_delete->forward_pointers[i];
451 }
452 }
453
454 Node* predecessor_lvl0 = node_to_delete->prev_pointer;
455 Node* successor_lvl0 = successor_node_ptr;
456
457 if (predecessor_lvl0) {
458 predecessor_lvl0->forward_pointers[0] = successor_lvl0;
459 }
460
461 if (successor_lvl0 != tail) {
462 assert(successor_lvl0 != nullptr && "Successor at L0 is null but not tail in erase");
463 successor_lvl0->prev_pointer = predecessor_lvl0;
464 } else {
465 tail->prev_pointer = predecessor_lvl0;
466 }
467
468 internal_delete_node(node_to_delete);
469 list_size--;
470
471 // Adjust current_max_level if necessary
472 while (current_max_level > 0 &&
473 head && static_cast<size_t>(current_max_level) < head->forward_pointers.size() &&
474 head->forward_pointers[current_max_level] == tail) {
475 current_max_level--;
476 }
477 return next_iter;
478 }
